22      public BattleMenu(NPC npcOpponent, Player player) throws DeathException {
23          this.random = new Random();
24          this.npcOpponent = npcOpponent;
25          this.player = player;
26          this.menuItems.add(new MenuItem("Attack", "Attack " + npcOpponent.getName() + "."));
27          this.menuItems.add(new MenuItem("Defend", "Defend against " + npcOpponent.getName() + "'s attack."));
28          this.menuItems.add(new MenuItem("Equip", "Equip an item"));
29          this.menuItems.add(new MenuItem("Unequip", "Unequip an item"));
30          this.menuItems.add(new MenuItem("View", "View details about your character"));
31          this.armour = player.getArmour();
32          this.damage = player.getDamage();
33          while (npcOpponent.getHealth() > 0 && player.getHealth() > 0) {
34              QueueProvider.offer("\nWhat is your choice?");
35              MenuItem selectedItem = displayMenu(this.menuItems);
36              testSelected(selectedItem);
37          }
38          if (player.getHealth() == 0) {
39              QueueProvider.offer("You died... Start again? (y/n)");
40              String reply = QueueProvider.take().toLowerCase();
41              while (!reply.startsWith("y") && !reply.startsWith("n")) {
42                  QueueProvider.offer("You died... Start again? (y/n)");
43                  reply = QueueProvider.take().toLowerCase();
44              }
45              if (reply.startsWith("y")) {
46                  throw new DeathException("restart");
47              } else if (reply.startsWith("n")) {
48                  throw new DeathException("close");
49              }
50          }  else if (npcOpponent.getHealth() == 0) {
51              int xp = npcOpponent.getXPGain();
52              this.player.setXP(this.player.getXP() + xp);
53              int oldLevel = this.player.getLevel();
54              int newLevel = (int) (0.075 * Math.sqrt(this.player.getXP()) + 1);
55              this.player.setLevel(newLevel);
56              //this.player.getLocation().removeMonster(npcOpponent);
57              this.player.setGold(this.player.getGold() + npcOpponent.getGold());
58              QueueProvider.offer("You killed a " + npcOpponent.getName() + "\nYou have gained " + xp + " XP and " + npcOpponent.getGold() + " gold");
59              if (oldLevel < newLevel) {
60                  QueueProvider.offer("You've are now level " + newLevel + "!");
61              }
62              CharacterChange cc = new CharacterChange();
63              cc.trigger(this.player, "kill", npcOpponent.getName());
64          }
65      }

112     private void testSelected(MenuItem m) {
113         switch (m.getKey()) {
114             case "attack": {
115                 mutateStats(1, 0.5);
116                 if (npcOpponent == null) {
117                     attack(player, monsterOpponent);
118                     attack(monsterOpponent, player);
119                 } else {
120                     attack(player, npcOpponent);
121                     attack(npcOpponent, player);
122                 }
123                 resetStats();
124                 break;
125             }
126             case "defend": {
127                 mutateStats(0.5, 1);
128                 if (npcOpponent == null) {
129                     QueueProvider.offer("\nYou get ready to defend against the " + monsterOpponent.getName() + ".");
130                     attack(player, monsterOpponent);
131                     attack(monsterOpponent, player);
132                 } else {
133                     QueueProvider.offer("\nYou get ready to defend against the " + npcOpponent.getName() + ".");
134                     attack(player, npcOpponent);
135                     attack(npcOpponent, player);
136                 }
137                 resetStats();
138                 break;
139             }
140             case "equip": {
141                 equip();
142                 break;
143             }
144             case "unequip": {
145                 unequip();
146                 break;
147             }
148             case "view": {
149                 viewStats();
150                 break;
151             }
152             default: {
153                 break;
154             }
155         }
156     }

158     private void attack(Entity attacker, Entity defender) {
159         if (attacker.getHealth() == 0) {
160             return;
161         }
162         double damage = attacker.getDamage();
163         double critCalc = random.nextDouble();
164         if (critCalc < attacker.getCritChance()) {
165             damage += damage;
166             QueueProvider.offer("Crit hit! Damage has been doubled!");
167         }
168         int healthReduction = (int) ((((3 * attacker.getLevel() / 50 + 2) * damage * damage / (defender.getArmour() + 1)/ 100) + 2) * (random.nextDouble() + 1));
169         defender.setHealth((defender.getHealth() - healthReduction));
170         if (defender.getHealth() < 0) {
171             defender.setHealth(0);
172         }
173         QueueProvider.offer(healthReduction + " damage dealt!");
174         if (attacker instanceof Player) {
175             QueueProvider.offer("The " + defender.getName() + "'s health is " + defender.getHealth());
176         } else {
177             QueueProvider.offer("Your health is " + defender.getHealth());
178         }
179     }

181     private void mutateStats(double damageMult, double armourMult) {
182         armour = player.getArmour();
183         damage = player.getDamage();
184         player.setArmour((int) (armour * armourMult));
185         player.setDamage(damage * damageMult);
186     }
187 
188     private void resetStats() {
189         player.setArmour(armour);
190         player.setDamage(damage);
191     }
192 
193     private void equip() {
194         player.printStorage();
195         QueueProvider.offer("What item do you want to use?");
196         String itemName = QueueProvider.take();
197         if (!itemName.equalsIgnoreCase("back")) {
198             player.equipItem(itemName);
199         }
200     }
201 
202     private void unequip() {
203         player.printEquipment();
204         QueueProvider.offer("What item do you want to unequip?");
205         String itemName = QueueProvider.take();
206         if (!itemName.equalsIgnoreCase("back")) {
207             player.dequipItem(itemName);
208         }
209     }

211     private void viewStats() {
212         QueueProvider.offer("\nWhat is your command?");
213         String input = QueueProvider.take();
214         switch (input) {
215             case "vs":
216             case "viewstats":
217                 player.getStats();
218                 break;
219             case "ve":
220             case "viewequipped":
221                 player.printEquipment();
222                 break;
223             case "vb":
224             case "viewbackpack":
225                 player.printStorage();
226                 break;
227             case "back":
228             case "exit":
229                 break;
230             default:
231                 viewStats();
232                 break;
233         }
234     }
