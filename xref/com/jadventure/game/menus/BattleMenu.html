11  public class BattleMenu extends Menus {

13      private Monster opponent;
14      private Player player;
15      private Random random;
16      private int armour;
17      private double damage;

19      public BattleMenu(Monster opponent, Player player) throws DeathException {
20          this.random = new Random();
21          this.opponent = opponent;
22          this.player = player;
23          this.menuItems.add(new MenuItem("Attack", "Attack " + opponent.monsterType + "."));
24          this.menuItems.add(new MenuItem("Defend", "Defend against " + opponent.monsterType + "'s attack."));
25          this.menuItems.add(new MenuItem("Equip", "Equip an item"));
26          this.menuItems.add(new MenuItem("Unequip", "Unequip an item"));
27          this.menuItems.add(new MenuItem("View", "View details about your character"));
28          this.armour = player.getArmour();
29          this.damage = player.getDamage();
30          while (opponent.getHealth() > 0 && player.getHealth() > 0) {
31              QueueProvider.offer("What is your choice?");
32              MenuItem selectedItem = displayMenu(this.menuItems);
33              testSelected(selectedItem);
34          }
35          if (player.getHealth() == 0) {
36              QueueProvider.offer("You died... Start again? (y/n)");
37              String reply = QueueProvider.take().toLowerCase();
38              while (!reply.startsWith("y") && !reply.startsWith("n")) {
39                  QueueProvider.offer("You died... Start again? (y/n)");
40                  reply = QueueProvider.take().toLowerCase();
41              }
42              if (reply.startsWith("y")) {
43                  throw new DeathException("restart");
44              } else if (reply.startsWith("n")) {
45                  throw new DeathException("close");
46              }
47          }  else if (opponent.getHealth() == 0) {
48              int xp = opponent.getXPGain();
49              this.player.setXP(this.player.getXP() + xp);
50              int oldLevel = this.player.getLevel();
51              int newLevel = (int) (0.075 * Math.sqrt(this.player.getXP()) + 1);
52              this.player.setLevel(newLevel);
53              this.player.getLocation().removeMonster(opponent);
54              this.player.setGold(this.player.getGold() + opponent.getGold());
55              QueueProvider.offer("You killed a " + opponent.monsterType + "\nYou have gained " + xp + " XP and " + opponent.getGold() + " gold");
56              if (oldLevel < newLevel) {
57                  QueueProvider.offer("You've are now level " + newLevel + "!");
58              }
59          }
60      }
61  
62      private void testSelected(MenuItem m) {
63          switch (m.getKey()) {
64              case "attack": {
65                  mutateStats(1, 0.5);
66                  attack(player, opponent);
67                  attack(opponent, player);
68                  resetStats();
69                  break;
70              }
71              case "defend": {
72                  QueueProvider.offer("You get ready to defend against the " + opponent.monsterType + ".");
73                  mutateStats(0.5, 1);
74                  attack(opponent, player);
75                  attack(player, opponent);
76                  resetStats();
77                  break;
78              }
79              case "equip": {
80                  equip();
81                  break;
82              }
83              case "unequip": {
84                  unequip();
85                  break;
86              }
87              case "view": {
88                  viewStats();
89                  break;
90              }
91              default: {
92                  break;
93              }
94          }
95      }
96  
97      private void attack(Entity attacker, Entity defender) {
98          if (attacker.getHealth() == 0) {
99              return;
100         }
101         double damage = attacker.getDamage();
102         double critCalc = random.nextDouble();
103         if (critCalc < attacker.getCritChance()) {
104             damage += damage;
105             QueueProvider.offer("Crit hit! Damage has been doubled!");
106         }
107         int healthReduction = (int) ((((3 * attacker.getLevel() / 50 + 2) * damage * damage / (defender.getArmour() + 1)/ 100) + 2) * (random.nextDouble() + 1));
108         defender.setHealth((defender.getHealth() - healthReduction));
109         if (defender.getHealth() < 0) {
110             defender.setHealth(0);
111         }
112         QueueProvider.offer(healthReduction + " damage dealt!");
113         if (attacker instanceof Player) {
114             Monster m = (Monster) defender;
115             QueueProvider.offer(m.monsterType + "' health is " + m.getHealth());
116         } else {
117             QueueProvider.offer("Your health is " + defender.getHealth());
118         }
119     }
120 
121     private void mutateStats(double damageMult, double armourMult) {
122         armour = player.getArmour();
123         damage = player.getDamage();
124         player.setArmour((int) (armour * armourMult));
125         player.setDamage(damage * damageMult);
126     }
127 
128     private void resetStats() {
129         player.setArmour(armour);
130         player.setDamage(damage);
131     }
132 
133     private void equip() {
134         player.printStorage();
135         QueueProvider.offer("What item do you want to use?");
136         String itemName = QueueProvider.take();
137         if (!itemName.equalsIgnoreCase("back")) {
138             player.equipItem(itemName);
139         }
140     }
141 
142     private void unequip() {
143         player.printEquipment();
144         QueueProvider.offer("What item do you want to unequip?");
145         String itemName = QueueProvider.take();
146         if (!itemName.equalsIgnoreCase("back")) {
147             player.dequipItem(itemName);
148         }
149     }
150 
151     private void viewStats() {
152         QueueProvider.offer("What is your command?");
153         String input = QueueProvider.take();
154         switch (input) {
155             case "vs":
156             case "viewstats":
157                 player.getStats();
158                 break;
159             case "ve":
160             case "viewequipped":
161                 player.printEquipment();
162                 break;
163             case "vb":
164             case "viewbackpack":
165                 player.printStorage();
166                 break;
167             case "back":
168             case "exit":
169                 break;
170             default:
171                 viewStats();
172                 break;
173         }
174     }
175 }
