40      public static void offer(String message) {
41          logger.debug("offer( " + message + " )");
42  
43          if (GameModeType.SERVER == mode) {
44              try {
45                  out = new DataOutputStream(server.getOutputStream());
46                  in = new DataInputStream(server.getInputStream());
47              } catch (IOException ioe) { 
48                  logger.debug("Inside offer( " + message + " )", ioe);
49              }
50          }
51  
52          if (GameModeType.SERVER == mode) {
53              sendToServer(message);
54          } else {
55              System.out.println(message);
56          }
57      }
58  
59      public static boolean sendToServer(String message) {
60          logger.debug("sendToServer( " + message + " )");
61          try {
62              out.writeUTF(message + "END");
63          } catch(SocketException se) { 
64              logger.debug("Inside  sendToServer( " + message + " )", se);
65              return false;
66          } catch(IOException ioe) { 
67              logger.debug("Inside  sendToServer( " + message + " )", ioe);
68              return false;
69          }
70          return true;
71      }
72  
73      public static String getInput(String message) {
74          logger.debug("getInput( " + message + " )");
75          String input = "";
76          try {
77              out.writeUTF(message + "END");
78              input = in.readUTF();
79          } catch(SocketException se) { 
80              logger.debug("Inside getInput( " + message + " )", se);
81              input = "error";
82          } catch(IOException ioe) { 
83              logger.debug("Inside getInput( " + message + " )", ioe);
84              input = "error";
85          }
86          return input;
87      }
