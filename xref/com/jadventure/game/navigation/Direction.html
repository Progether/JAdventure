3   /**
4    * An enumeration with four properties: A description for in-game text and three integers.
5    * The integers are added to a coordinate to get the new direction (i.e. to get a north coordinate
6    * you add 1 to the y property of a coordinate).
7    */
8   public enum Direction {
9       NORTH("To the North", 0, 1, 0),
10      SOUTH("To the South", 0, -1, 0),
11      EAST("To the East", 1, 0, 0),
12      WEST("To the West", -1, 0, 0),
13      DOWN("Heading down", 0, 0, -1),
14      UP("Heading up", 0, 0, 1);
15  
16      private final String description;
17      private final int dx;
18      private final int dy;
19      private final int dz;
20  
21      private Direction(String description, int dx, int dy, int dz) {
22          this.description = description;
23          this.dx = dx;
24          this.dy = dy;
25          this.dz = dz;
26      }
27  
28      public String getDescription() {
29          return description;
30      }
31  
32      public int getDx() {
33          return dx;
34      }
35  
36      public int getDy() {
37          return dy;
38      }
39  
40      public int getDz() {
41          return dz;
42      }
43  }
