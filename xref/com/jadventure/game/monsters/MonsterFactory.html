1   package com.jadventure.game.monsters;
2   
3   import com.jadventure.game.entities.Player;
4   
5   import java.util.Random;
6   
7   /**
8    * The MonsterFactory generates random monsters appropriately according
9    * to the player's level and location
10   */
11  public class MonsterFactory {
12      Random random = new Random();
13   
14      public Monster generateMonster(Player player) {
15          int randomInt = random.nextInt(5) + 1;
16          if (randomInt <= player.getLocation().getDangerRating()) {
17              switch (player.getLocationType()) {
18                  case FOREST:
19                      return getForestMonster(player.getLevel());
20                  case SWAMP:
21                      return getSwampMonster(player.getLevel());
22                  case MOUNTAIN:
23                      return getMountainMonster(player.getLevel());   
24                  case CAVE:
25                      return getCaveMonster(player.getLevel());
26                  case PLAINS:
27                      return getPlainsMonster(player.getLevel());
28                  default: // any non-hostile location
29                      return null;
30              }
31          } else {
32              return null;
33          }
34      }
35  
36      private Monster getForestMonster(int playerLevel) {
37          int randomInt = random.nextInt(4);
38          if (randomInt == 0)
39          	return new Bugbear(playerLevel);
40          else if (randomInt == 1)
41          	return new Troll(playerLevel);
42          else
43          	return new Goblin(playerLevel);
44      }
45      
46      private Monster getSwampMonster(int playerLevel) {
47      	int randomInt = random.nextInt(2);
48          return (randomInt == 1) ? new Goblin(playerLevel) : new Troll(playerLevel);
49      }
50      
51      private Monster getMountainMonster(int playerLevel) {
52      	int randomInt = random.nextInt(4);
53          if (randomInt == 0)
54              return new Giant(playerLevel);
55          else if (randomInt == 1)
56              return new Troll(playerLevel);
57          else if (randomInt == 2)
58              return new Wolf(playerLevel);
59          else
60              return new Skeleton(playerLevel);
61      }
62  
63      private Monster getCaveMonster(int playerLevel) {
64      	int randomInt = random.nextInt(4);
65          if (randomInt == 0)
66              return new Troll(playerLevel);
67          else if (randomInt == 1)
68              return new Skeleton(playerLevel);
69          else
70              return new Goblin(playerLevel);
71      }
72  
73      private Monster getPlainsMonster(int playerLevel) {
74      	int randomInt = random.nextInt(2);
75      	return (randomInt == 1) ? new Bugbear(playerLevel) : new Goblin(playerLevel);
76      }
77  }
