105     public static Player load(String name) {
106         player = new Player();
107         JsonParser parser = new JsonParser();
108         String fileName = getProfileFileName(name);
109         try {
110             Reader reader = new FileReader(fileName);
111             JsonObject json = parser.parse(reader).getAsJsonObject();
112             player.setName(json.get("name").getAsString());
113             player.setHealthMax(json.get("healthMax").getAsInt());
114             player.setHealth(json.get("health").getAsInt());
115             player.setArmour(json.get("armour").getAsInt());
116             player.setDamage(json.get("damage").getAsInt());
117             player.setLevel(json.get("level").getAsInt());
118             player.setXP(json.get("xp").getAsInt());
119             player.setStrength(json.get("strength").getAsInt());
120             player.setIntelligence(json.get("intelligence").getAsInt());
121             player.setDexterity(json.get("dexterity").getAsInt());
122             player.setLuck(json.get("luck").getAsInt());
123             player.setStealth(json.get("stealth").getAsInt());
124             player.setCurrentCharacterType(json.get("type").getAsString());
125             HashMap<String, Integer> charLevels = new Gson().fromJson(json.get("types"), new TypeToken<HashMap<String, Integer>>(){}.getType());
126             player.setCharacterLevels(charLevels);
127             if (json.has("equipment")) {
128                 Map<String, EquipmentLocation> locations = new HashMap<>();
129                 locations.put("head", EquipmentLocation.HEAD);
130                 locations.put("chest", EquipmentLocation.CHEST);
131                 locations.put("leftArm", EquipmentLocation.LEFT_ARM);
132                 locations.put("leftHand", EquipmentLocation.LEFT_HAND);
133                 locations.put("rightArm", EquipmentLocation.RIGHT_ARM);
134                 locations.put("rightHand", EquipmentLocation.RIGHT_HAND);
135                 locations.put("bothHands", EquipmentLocation.BOTH_HANDS);
136                 locations.put("bothArms", EquipmentLocation.BOTH_ARMS);
137                 locations.put("legs", EquipmentLocation.LEGS);
138                 locations.put("feet", EquipmentLocation.FEET);
139                 HashMap<String, String> equipment = new Gson().fromJson(json.get("equipment"), new TypeToken<HashMap<String, String>>(){}.getType());
140                Map<EquipmentLocation, Item> equipmentMap = new HashMap<>();
141                for(Map.Entry<String, String> entry : equipment.entrySet()) {
142                    EquipmentLocation el = locations.get(entry.getKey());
143                    Item i = itemRepo.getItem(entry.getValue());
144                    equipmentMap.put(el, i);
145                }
146                player.setEquipment(equipmentMap);
147             }
148             if (json.has("items")) {
149                 HashMap<String, Integer> items = new Gson().fromJson(json.get("items"), new TypeToken<HashMap<String, Integer>>(){}.getType());
150                 ArrayList<ItemStack> itemList = new ArrayList<>();
151                 for (Map.Entry<String, Integer> entry : items.entrySet()) {
152                     String itemID = entry.getKey();
153                     int amount = entry.getValue();
154                     Item item = itemRepo.getItem(itemID);
155                     ItemStack itemStack = new ItemStack(amount, item);
156                     itemList.add(itemStack);
157                 }
158                 float maxWeight = (float)Math.sqrt(player.getStrength()*300);
159                 player.setStorage(new Storage(maxWeight, itemList));
160             }
161             Coordinate coordinate = new Coordinate(json.get("location").getAsString());
162             LocationManager.getInstance(player.getName());
163             player.setLocation(LocationManager.getLocation(coordinate));
164             reader.close();
165             setUpCharacterLevels();
166         } catch (FileNotFoundException ex) {
167             QueueProvider.offer( "Unable to open file '" + fileName + "'.");
168         } catch (IOException ex) {
169             ex.printStackTrace();
170         }
171 
172         return player;
173     }

178     public static Player getInstance(String playerClass){
179         player = new Player();
180         JsonParser parser = new JsonParser();
181         String fileName = "json/npcs.json";
182         try {
183             Reader reader = new FileReader(fileName);
184             JsonObject npcs = parser.parse(reader).getAsJsonObject().get("npcs").getAsJsonObject();
185             JsonObject json = new JsonObject();
186             for (Map.Entry<String, JsonElement> entry : npcs.entrySet()) {
187                 if (entry.getKey().equals(playerClass)) {
188                     json = entry.getValue().getAsJsonObject();
189                 }
190             }
191 
192             player.setName(json.get("name").getAsString());
193             player.setHealthMax(json.get("healthMax").getAsInt());
194             player.setHealth(json.get("health").getAsInt());
195             player.setArmour(json.get("armour").getAsInt());
196             player.setDamage(json.get("damage").getAsInt());
197             player.setLevel(json.get("level").getAsInt());
198             player.setXP(json.get("xp").getAsInt());
199             player.setStrength(json.get("strength").getAsInt());
200             player.setIntelligence(json.get("intelligence").getAsInt());
201             player.setDexterity(json.get("dexterity").getAsInt());
202             setUpVariables(player);
203             JsonArray items = json.get("items").getAsJsonArray();
204             for (JsonElement item : items) {
205                 player.addItemToStorage(itemRepo.getItem(item.getAsString()));
206             }
207             Random rand = new Random();
208             int luck = rand.nextInt(3) + 1;
209             player.setLuck(luck);
210             player.setStealth(json.get("stealth").getAsInt());
211             player.setIntro(json.get("intro").getAsString());
212             if (player.getName().equals("Recruit")) {
213                 player.type = "Recruit";
214             } else if (player.getName().equals("Sewer Rat")) {
215                 player.type = "Sewer Rat";
216             } else {
217                 QueueProvider.offer("Not a valid class");
218             }
219             reader.close();
220             setUpCharacterLevels();
221         } catch (FileNotFoundException ex) {
222             QueueProvider.offer( "Unable to open file '" + fileName + "'.");
223         } catch (IOException ex) {
224             ex.printStackTrace();
225         }
226 
227         return player;
228     }

269     public void save() {
270         Gson gson = new Gson();
271         JsonObject jsonObject = new JsonObject();
272         jsonObject.addProperty("name", getName());
273         jsonObject.addProperty("healthMax", getHealthMax());
274         jsonObject.addProperty("health", getHealthMax());
275         jsonObject.addProperty("armour", getArmour());
276         jsonObject.addProperty("damage", getDamage());
277         jsonObject.addProperty("level", getLevel());
278         jsonObject.addProperty("xp", getXP());
279         jsonObject.addProperty("strength", getStrength());
280         jsonObject.addProperty("intelligence", getIntelligence());
281         jsonObject.addProperty("dexterity", getDexterity());
282         jsonObject.addProperty("luck", getLuck());
283         jsonObject.addProperty("stealth", getStealth());
284         jsonObject.addProperty("weapon", getWeapon());
285         jsonObject.addProperty("type", getCurrentCharacterType());
286         HashMap<String, Integer> items = new HashMap<String, Integer>();
287         for (ItemStack item : getStorage().getItemStack()) {
288             items.put(item.getItem().getId(), item.getAmount());
289         }
290         JsonElement itemsJsonObj = gson.toJsonTree(items);
291         jsonObject.add("items", itemsJsonObj);
292         Map<EquipmentLocation, String> locations = new HashMap<>();
293         locations.put(EquipmentLocation.HEAD, "head");
294         locations.put(EquipmentLocation.CHEST, "chest");
295         locations.put(EquipmentLocation.LEFT_ARM, "leftArm");
296         locations.put(EquipmentLocation.LEFT_HAND, "leftHand");
297         locations.put(EquipmentLocation.RIGHT_ARM, "rightArm");
298         locations.put(EquipmentLocation.RIGHT_HAND, "rightHand");
299         locations.put(EquipmentLocation.BOTH_HANDS, "BothHands");
300         locations.put(EquipmentLocation.BOTH_ARMS, "bothArms");
301         locations.put(EquipmentLocation.LEGS, "legs");
302         locations.put(EquipmentLocation.FEET, "feet");
303         HashMap<String, String> equipment = new HashMap<>();
304         Item hands = itemRepo.getItem("hands");
305         for (Map.Entry<EquipmentLocation, Item> item : getEquipment().entrySet()) {
306             if (item.getKey() != null && !hands.equals(item.getValue()) && item.getValue() != null) {
307                 equipment.put(locations.get(item.getKey()), item.getValue().getId());
308             }
309         }
310         JsonElement equipmentJsonObj = gson.toJsonTree(equipment);
311         jsonObject.add("equipment", equipmentJsonObj);
312         JsonElement typesJsonObj = gson.toJsonTree(getCharacterLevels());
313         jsonObject.add("types", typesJsonObj);
314         Coordinate coordinate = getLocation().getCoordinate();
315         String coordinateLocation = coordinate.x+","+coordinate.y+","+coordinate.z;
316         jsonObject.addProperty("location", coordinateLocation);
317 
318         String fileName = getProfileFileName(getName());
319         new File(fileName).getParentFile().mkdirs();
320         try {
321             Writer writer = new FileWriter(fileName);
322             gson.toJson(jsonObject, writer);
323             writer.close();
324             LocationManager.writeLocations(getName());
325             QueueProvider.offer("\nYour game data was saved.");
326         } catch (IOException ex) {
327             QueueProvider.offer("\nUnable to save to file '" + fileName + "'.");
328         }
329     }

366     public void dropItem(String itemName) {
367         List<Item> itemMap = searchItem(itemName, getStorage());
368         if (itemMap.isEmpty()) {
369             itemMap = searchEquipment(itemName, getEquipment());
370         }
371         if (!itemMap.isEmpty()) {
372             Item item = itemMap.get(0);
373             Item itemToDrop = itemRepo.getItem(item.getId());
374             Item weapon = itemRepo.getItem(getWeapon());
375             String wName = weapon.getName();
376 
377             if (itemName.equals(wName)) {
378                 dequipItem(wName);
379             }
380             removeItemFromStorage(itemToDrop);
381             location.addItem(itemToDrop);
382             QueueProvider.offer(item.getName() + " dropped");
383         }
384     }

386     public void equipItem(String itemName) {
387         List<Item> items = searchItem(itemName, getStorage());
388         if (!items.isEmpty()) {
389             Item item = items.get(0);
390             if (getLevel() >= item.getLevel()) {
391                 Map<String, String> change = equipItem(item.getPosition(), item);
392                 QueueProvider.offer(item.getName()+ " equipped");
393                 printStatChange(change);
394             } else {
395                 QueueProvider.offer("You do not have the required level to use this item");
396             }
397         } else {
398             QueueProvider.offer("You do not have that item");
399         }
400     }

412     private void printStatChange(Map<String, String> stats) {
413          Set<Entry<String, String>> set = stats.entrySet();
414          Iterator<Entry<String, String>> iter = set.iterator();
415          while (iter.hasNext()) {
416               Entry<String, String> me = iter.next();
417               double value = Double.parseDouble((String) me.getValue());
418               switch ((String) me.getKey()) {
419                   case "damage": {
420                           if (value >= 0.0) {
421                               QueueProvider.offer(me.getKey() + ": " + this.getDamage() + " (+" + me.getValue() + ")");
422                           } else {
423                               QueueProvider.offer(me.getKey() + ": " + this.getDamage() + " (" + me.getValue() + ")");
424                           }
425                           break;
426                     }
427                     case "health": {
428                           if (value >= 0) {
429                               QueueProvider.offer(me.getKey() + ": " + this.getHealth() + " (+" + me.getValue() + ")");
430                           } else {
431                               QueueProvider.offer(me.getKey() + ": " + this.getHealth() + " (" + me.getValue() + ")");
432                           }
433                           break;
434                     }
435                     case "armour": {
436                           if (value >= 0) {
437                               QueueProvider.offer(me.getKey() + ": " + this.getArmour() + " (+" + me.getValue() + ")");
438                           } else {
439                               QueueProvider.offer(me.getKey() + ": " + this.getArmour() + " (" + me.getValue() + ")");
440                           }
441                           break;
442                     }
443                     case "maxHealth": {
444                           if (value  >= 0) {
445                               QueueProvider.offer(me.getKey() + ": " + this.getHealthMax() + " (+" + me.getValue() + ")");
446                           } else {
447                               QueueProvider.offer(me.getKey() + ": " + this.getHealthMax() + " (" + me.getValue() + ")");
448                           }
449                           break;
450                     }
451               }
452          }
453     }

480     public void attack(String opponentName) throws DeathException {
481         Monster monsterOpponent = null;
482         NPC npcOpponent = null;
483         List<Monster> monsters = getLocation().getMonsters();
484         List<NPC> npcs = getLocation().getNPCs();
485         for (int i = 0; i < monsters.size(); i++) {
486              if (monsters.get(i).monsterType.equalsIgnoreCase(opponentName)) {
487                  monsterOpponent = monsters.get(i);
488              }
489         }
490         for (int i=0; i < npcs.size(); i++) {
491             if (npcs.get(i).getName().equalsIgnoreCase(opponentName)) {
492                 npcOpponent = npcs.get(i);
493             }
494         }
495         if (monsterOpponent != null) {
496             monsterOpponent.setName(monsterOpponent.monsterType);
497             new BattleMenu(monsterOpponent, this);
498         } else if (npcOpponent != null) {
499             new BattleMenu(npcOpponent, this);
500         } else {
501              QueueProvider.offer("Opponent not found");
502         }
503     }
