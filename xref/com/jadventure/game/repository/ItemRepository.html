42              if (id == null || id.trim().length() == 0) {
43                  return null;

45  	    if (!itemMap.containsKey(id)) {

60      // Load all items, from the given file
61      protected void load(JsonReader reader) {
62          JsonObject jsonItems = new JsonObject();
63          try {
64              JsonParser parser = new JsonParser();
65              JsonObject json = parser.parse(reader).getAsJsonObject();
66              jsonItems = json.get("items").getAsJsonObject();
67              reader.close();
68          } catch (FileNotFoundException ex) {
69              ex.printStackTrace();
70          } catch (IOException ex) {
71              ex.printStackTrace();
72          }
73          
74          for (Map.Entry<String, JsonElement> entry : jsonItems.entrySet()) {
75              String id = entry.getKey();
76              JsonObject itemData = entry.getValue().getAsJsonObject();
77              String type = itemData.get("type").getAsString();
78              String name = itemData.get("name").getAsString();
79              String description = itemData.get("description").getAsString();
80              EquipmentLocation position = EQUIPMENT_POSITION_MAP.get(itemData.get("position").getAsString());
81              int level = itemData.get("level").getAsInt();
82              JsonObject sProps = itemData.get("properties").getAsJsonObject();
83              Map<String, Integer> properties = new TreeMap<>();
84              for (Map.Entry<String, JsonElement> entry2 : sProps.entrySet()) {
85                  Integer propValue = entry2.getValue().getAsInt();
86                  properties.put(entry2.getKey(), propValue);
87              }
88  
89              addItem(new Item(id, type, name, description, position, level, properties));
90              
91          }
92      }

100     public static ItemRepository createRepo() {
101         if (itemRepository == null) {
102             File file = new File(new File(System.getProperty("user.dir")), "json");
103 
104             File dataFile = new File(new File(file, "original_data"), "items.json");
105             if (! dataFile.exists()) {
106             	throw new RuntimeException("File '" + dataFile + "' does not exist.");
107             }
108 
109             itemRepository = new ItemRepository();
110             itemRepository.load(dataFile);
111         }
112         return itemRepository;
113     }

120     public void store(JsonWriter writer) {
121         GsonBuilder bldr = new GsonBuilder().setPrettyPrinting();
122         Gson gson = bldr.create();
123         Map<String, Map<String, Item>> root = new TreeMap<>();
124         root.put("items", itemMap);
125         gson.toJson(root, Map.class, writer);
126     }

144     public Item getRandomItem(String start, int level) {
145         Random rand = new Random();
146         int chance = rand.nextInt(100);
147         if (chance < 70) {
148             Item item = null;
149             do {
150                 item = getRandom(start);
151             } while (item.getLevel() > level);
152             return item;
153         } else {
154             return getRandom(start);
155         }
156     }
157 
158     private Item getRandom(String start) {
159         Random rand = new Random();
160         Item item = null;
161         do {
162             int itemIndex = rand.nextInt(itemMap.size() - 2);
163             List<Item> items = new ArrayList<>(itemMap.values());
164             item = items.get(itemIndex + 2); // avoids empty and hands items
165         } while (!item.getId().startsWith(start));
166         return item;
167     }
